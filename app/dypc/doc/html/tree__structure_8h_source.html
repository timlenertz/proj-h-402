68 void tree_structure<Splitter, Levels, PointsContainer>::downsample_points_in_node_(float ratio, const node& nd, const cuboid& cub, PointsContainer& output, unsigned depth) {
69  if(nd.is_leaf() || !downsampling_maximal_number_of_points_ || nd.number_of_points() <= downsampling_maximal_number_of_points_) {
70  float area = cub.area();
71  if(downsampling_mode_ == downsampling_mode::random) random_downsampling(nd.points_begin(), nd.points_end(), ratio, output);
72  else if(downsampling_mode_ == downsampling_mode::uniform) uniform_downsampling(nd.points_begin(), nd.points_end(), ratio, area, output, node_uniform_downsampling_previous_results_[&nd]);
73 
74  } else {
75  for(std::ptrdiff_t i = 0; i < Splitter::number_of_node_children; ++i) {
76  auto child_cuboid = Splitter::node_child_cuboid(i, cub, nd.get_points_information(), depth);
77  downsample_points_in_node_(ratio, nd.child(i), child_cuboid, output, depth + 1);
78  }
79  }
80 }

97 void tree_structure<Splitter, Levels, PointsContainer>::load_model_(model& mod, const cuboid& cub, unsigned depth) {
98  unload_model_();
99  root_cuboid_ = cub;
100 
101  PointsContainer all_points_unordered;
102  progress("Collecting points from model...", mod.number_of_points(), [&]() {
103  for(const auto& pt : mod) {
104  if(! cub.in_range(pt)) continue;
105  all_points_unordered.push_back(pt);
106  increment_progress();
107  }
108  });
109  progress("Adding points with info to tree...", all_points_unordered.size(), [&]() {
110  root_.add_points_with_information(all_points_unordered, root_cuboid_, depth, leaf_capacity_);
111  });
112 
113  root_.move_out_points(all_points_[0]);
114  root_.finalize_move_out(all_points_[0]);
115 
116 
117  const auto& points = all_points_[0];
118  float ratio = 1.0 / mipmap_factor_;
119  for(int lvl = 1; lvl < Levels; ++lvl) {
120  PointsContainer downsampled;
121 
122  downsample_points_in_node_(ratio, root_, root_cuboid_, downsampled, depth);
123 
124  progress("Adding downsampled points, level " + std::to_string(lvl) + "...", downsampled.size(), [&]() {
125  auto& level_points = all_points_[lvl];
126  std::size_t c = 0;
127  for(const point& pt : downsampled) {
128  root_.add_higher_level_point(lvl, pt, root_cuboid_, depth, leaf_capacity_);
129  increment_progress();
130  }
131  root_.move_out_points(level_points, lvl);
132  root_.finalize_move_out(level_points, lvl);
133  });
134 
135  ratio /= mipmap_factor_;
136  }
137 
138  node_uniform_downsampling_previous_results_.clear();
139 }
